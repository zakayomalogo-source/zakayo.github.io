🎨 2. CSS — Muonekano wa Tovuti

Njia za Kuandika CSS

/* 1. CSS ya nje (bora zaidi) */ <link rel="stylesheet" href="style.css"> /* 2. CSS ndani ya HTML */ <style> body { color: white; } </style> /* 3. CSS moja kwa moja */ <p style="color:red">Maandishi</p>

Selectors

/* Selector ya tag */ p { color: white; } /* Selector ya class */ .kitufe { background: red; } /* Selector ya id */ #kichwa { font-size: 50px; } /* Selector ya watoto */ nav a { color: white; } /* Selector ya hover */ button:hover { opacity: 0.8; } /* Selector ya kwanza */ li:first-child { color: red; }

Box Model

div { /* Upana na urefu */ width: 300px; height: 200px; /* Nafasi ndani */ padding: 20px; padding-top: 10px; /* Mipaka */ border: 2px solid red; border-radius: 10px; /* Nafasi nje */ margin: 20px; margin: 10px 20px; /* juu/chini kushoto/kulia */ /* Rangi */ background-color: blue; background: linear-gradient(red, blue); }

Maandishi

p { color: white; font-size: 16px; font-family: Arial, sans-serif; font-weight: bold; text-align: center; text-decoration: underline; line-height: 1.6; letter-spacing: 2px; text-transform: uppercase; }

Flexbox

.sanduku { display: flex; /* Mwelekeo */ flex-direction: row; /* au column */ /* Upangaji mlalo */ justify-content: center; /* space-between, space-around, flex-start, flex-end */ /* Upangaji wima */ align-items: center; /* flex-start, flex-end, stretch */ /* Pengo kati ya vitu */ gap: 20px; /* Wrap */ flex-wrap: wrap; }

Grid

.grid { display: grid; /* Nguzo 3 sawa */ grid-template-columns: 1fr 1fr 1fr; /* Au kutumia repeat */ grid-template-columns: repeat(3, 1fr); /* Responsive */ grid-template-columns: repeat(auto-fit, minmax(200px, 1fr)); /* Pengo */ gap: 20px; /* Safu */ grid-template-rows: auto; }

Position

div { /* Kawaida */ position: static; /* Inabadilika kutoka mahali pake */ position: relative; top: 10px; left: 20px; /* Imewekwa mahali fulani */ position: absolute; top: 0; right: 0; /* Inabaki ukisogea */ position: fixed; bottom: 20px; /* Kama fixed lakini ndani ya parent */ position: sticky; top: 0; }

Animations

/* Transition */ button { transition: all 0.3s ease; } button:hover { transform: scale(1.1); opacity: 0.8; } /* Animation */ @keyframes panda { from { opacity: 0; } to { opacity: 1; } } div { animation: panda 1s ease; } /* Transform */ div { transform: rotate(45deg); transform: translateX(100px); transform: scale(1.5); }

Media Queries (Responsive)

/* Simu (chini ya 768px) */ @media (max-width: 768px) { .grid { grid-template-columns: 1fr; } h1 { font-size: 24px; } } /* Tablet (768px - 1024px) */ @media (min-width: 768px) and (max-width: 1024px) { .grid { grid-template-columns: 1fr 1fr; } } /* Kompyuta kubwa */ @media (min-width: 1024px) { .grid { grid-template-columns: repeat(3, 1fr); } }

CSS Variables

/* Tangaza variables */ :root { --rangi-kuu: #e94560; --rangi-giza: #1a1a2e; --rangi-kati: #0f3460; --fonti: 'Arial', sans-serif; --nafasi: 20px; } /* Tumia variables */ body { background: var(--rangi-giza); font-family: var(--fonti); } button { background: var(--rangi-kuu); padding: var(--nafasi); }

⚡ 3. JavaScript — Uhai wa Tovuti

Variables

// var — zamani (epuka) var jina = "Zakayo"; // let — inabadilika let umri = 25; umri = 26; // sawa // const — haibadiliki const nchi = "Tanzania"; // nchi = "Kenya"; // KOSA!

Data Types

// String (maandishi) let jina = "Zakayo"; let salamu = 'Habari'; // Number (nambari) let umri = 25; let bei = 99.99; // Boolean (kweli/uongo) let ni_developer = true; let amechoka = false; // Array (orodha) let ujuzi = ["HTML", "CSS", "JS"]; // Object (kitu) let mtu = { jina: "Zakayo", umri: 25, nchi: "Tanzania" }; // Null na Undefined let tupu = null; let haijawekwa = undefined;

Functions

// Function ya kawaida function salamu(jina) { return "Habari " + jina; } // Arrow function const salamu = (jina) => { return "Habari " + jina; }; // Arrow fupi const salamu = jina => "Habari " + jina; // Function yenye default function piga(simu = "0700000000") { console.log("Kupiga: " + simu); } // Kutumia console.log(salamu("Zakayo")); piga();

Masharti (Conditions)

// If/else if (umri >= 18) { console.log("Mtu mzima"); } else if (umri >= 13) { console.log("Kijana"); } else { console.log("Mtoto"); } // Ternary operator let hali = umri >= 18 ? "Mzima" : "Mdogo"; // Switch switch (siku) { case "Jumatatu": console.log("Wiki inaanza"); break; case "Ijumaa": console.log("Wiki inaisha"); break; default: console.log("Siku nyingine"); }

Loops (Mizunguko)

// For loop for (let i = 0; i < 5; i++) { console.log(i); } // While loop let i = 0; while (i < 5) { console.log(i); i++; } // For...of (array) let ujuzi = ["HTML", "CSS", "JS"]; for (let u of ujuzi) { console.log(u); } // For...in (object) let mtu = { jina: "Zakayo", umri: 25 }; for (let key in mtu) { console.log(key + ": " + mtu[key]); } // forEach ujuzi.forEach(u => console.log(u));

Array Methods

let nambari = [1, 2, 3, 4, 5]; // map — badilisha kila kitu let maradufu = nambari.map(n => n * 2); // [2, 4, 6, 8, 10] // filter — chagua baadhi let kubwa = nambari.filter(n => n > 3); // [4, 5] // find — tafuta moja let tatu = nambari.find(n => n === 3); // 3 // reduce — hesabu jumla let jumla = nambari.reduce((a, b) => a + b, 0); // 15 // includes — angalia kama kipo nambari.includes(3); // true // push/pop — ongeza/ondoa mwishoni nambari.push(6); // [1,2,3,4,5,6] nambari.pop(); // ondoa mwisho // splice — ondoa mahali fulani nambari.splice(2, 1); // ondoa index 2 // join — unganisha nambari.join(", "); // "1, 2, 3, 4, 5"

DOM Manipulation

// Pata elementi const el = document.getElementById("id"); const els = document.querySelectorAll(".class"); const el2 = document.querySelector(".class"); // Badilisha maudhui el.innerHTML = "<b>Maandishi</b>"; el.textContent = "Maandishi tu"; // Badilisha CSS el.style.color = "red"; el.style.display = "none"; // Ongeza/Ondoa class el.classList.add("active"); el.classList.remove("active"); el.classList.toggle("active"); // Unda elementi mpya const div = document.createElement("div"); div.innerHTML = "Habari!"; document.body.appendChild(div); // Ondoa elementi el.remove();

Events

// Click button.addEventListener("click", function() { console.log("Imebonyezwa!"); }); // Arrow function button.addEventListener("click", () => { console.log("Imebonyezwa!"); }); // Aina za events element.addEventListener("click", fn); element.addEventListener("mouseover", fn); element.addEventListener("mouseout", fn); element.addEventListener("keypress", fn); element.addEventListener("keydown", fn); element.addEventListener("submit", fn); element.addEventListener("change", fn); element.addEventListener("load", fn); element.addEventListener("scroll", fn); // Event kwenye HTML <button onclick="fanya()">Bonyeza</button>

Async/Await na Fetch

// Fetch data kutoka API async function pataDana() { try { const res = await fetch("https://api.example.com"); const data = await res.json(); console.log(data); } catch (kosa) { console.log("Kosa: " + kosa); } } // POST request async function tumaDana(taarifa) { const res = await fetch("/api/data", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify(taarifa) }); return await res.json(); } pataDana();

Local Storage

// Hifadhi localStorage.setItem("jina", "Zakayo"); // Soma const jina = localStorage.getItem("jina"); // Futa moja localStorage.removeItem("jina"); // Futa yote localStorage.clear(); // Hifadhi object const mtu = { jina: "Zakayo", umri: 25 }; localStorage.setItem("mtu", JSON.stringify(mtu)); // Soma object const data = JSON.parse( localStorage.getItem("mtu"));

MongoDB na Mongoose

const mongoose = require("mongoose"); // Unganisha mongoose.connect("mongodb+srv://..."); // Schema const MtumiziSchema = new mongoose.Schema({ jina: String, email: { type: String, required: true }, umri: Number, tarehe: { type: Date, default: Date.now } }); // Model const Mtumiaji = mongoose.model( "Mtumiaji", MtumiziSchema); // CRUD Operations // Create const mpya = new Mtumiaji({ jina: "Zakayo" }); await mpya.save(); // Read const wote = await Mtumiaji.find(); const mmoja = await Mtumiaji.findById(id); // Update await Mtumiaji.findByIdAndUpdate(id, {jina:"Mpya"}); // Delete await Mtumiaji.findByIdAndDelete(id);
